CREATE TABLE allimtalk.mts_atalk_msg (
										tran_pr varchar(16) NOT NULL,
										tran_refkey varchar(20) NULL,
										tran_id varchar(20) NULL,
										tran_sender_key varchar(40) NULL,
										tran_tmpl_cd varchar(30) NOT NULL,
										tran_button varchar(2000) NULL,
										tran_phone varchar(15) NOT NULL,
										tran_msg varchar(2000) NOT NULL,
										tran_date timestamp NOT NULL,
										tran_type int4 NOT NULL,
										tran_status bpchar(1) NOT NULL DEFAULT '1'::bpchar,
										tran_senddate timestamp NULL,
										tran_reportdate timestamp NULL,
										tran_rsltdate timestamp NULL,
										tran_rslt varchar(5) NULL,
										tran_replace_msg varchar(2000) NOT NULL,
										tran_subject varchar(100) NULL,
										tran_callback varchar(15) NOT NULL,
										tran_replace_type bpchar(1) NULL DEFAULT 'N'::bpchar,
										tran_etc2 varchar(160) NULL,
										tran_etc1 varchar(160) NULL,
										tran_etc3 varchar(160) NULL,
										tran_etc4 varchar(160) NULL,
										tran_end_telco varchar(8) NULL,
										tran_log bpchar(1) NULL DEFAULT 'N'::bpchar,
										tran_grpseq int4 NULL,
										tran_title varchar(500) NULL,
										tran_nation_phone varchar(5) NULL DEFAULT '82'::character varying,
										tran_supplement varchar(4000) NULL,
										tran_price int8 NULL,
										tran_currency varchar(3) NULL,
										tran_header varchar(32) NULL,
										tran_attachment varchar(4000) NULL,
										CONSTRAINT "MTS_ATALK_MSG_PK" PRIMARY KEY (tran_pr)
);

COMMENT ON COLUMN allimtalk.mts_atalk_msg.tran_pr IS '메시지 Key 고유값 (시퀀스 생성 필요) ';

COMMENT ON COLUMN allimtalk.mts_atalk_msg.tran_tmpl_cd IS '알림톡 템플릿 코드';
COMMENT ON COLUMN allimtalk.mts_atalk_msg.tran_button IS '버튼 제목 , 링크(최대 5개까지)';
COMMENT ON COLUMN allimtalk.mts_atalk_msg.tran_phone IS '수신번호(G/W 전달값)';
COMMENT ON COLUMN allimtalk.mts_atalk_msg.tran_msg IS '전송 메세지(G/W 전달값) 영문 1000자 , 한글 1000자 ';
COMMENT ON COLUMN allimtalk.mts_atalk_msg.tran_date IS '전송요청시간';

COMMENT ON COLUMN allimtalk.mts_atalk_msg.tran_senddate IS '전송시간';

COMMENT ON COLUMN allimtalk.mts_atalk_msg.tran_rslt IS 'Report 결과 코드';

COMMENT ON COLUMN allimtalk.mts_atalk_msg.tran_subject IS '제목(LMS 전송시 사용)';

COMMENT ON COLUMN allimtalk.mts_atalk_msg.tran_replace_type IS '전환전송 타입 S sms L lms mms N none ';
